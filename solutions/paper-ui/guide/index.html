# 引用某个组件


 




 











 









<script>
import { PButton } from 'vue-paper-ui';

export default {
  name: 'app',
  components: {
    PButton
  },
  methods: {
    handleClick() {
      alert('hi~there');
    }
  }
}
</script>

<template>
  <div>
    <PButton @click="handleClick">Button</PButton>
  </div>
</template>
